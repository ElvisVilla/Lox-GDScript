class Weapon {

	var price: int = 200
	var durability: int = 32
	
	func init(price: int, durability: int) {
		self.price = price
		self.durability = durability
	}
	
	func showMessage(message, age: int = 30) {
		var index = 0
		while(index < 10000) {
			print(message + " with age: " + age + "Current Index: " + index)
			index = index + 1
		}
	}
}


var weapon = Weapon(500, 500)
weapon.showMessage("My name is Elvis")
